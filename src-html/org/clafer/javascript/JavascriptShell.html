025/**
026 * The Javascript CLI.
027 *
028 * @author jimmy
029 */
030public class JavascriptShell {
031
032    // The Javascript engine.
033    private final ScriptEngine engine = Javascript.newEngine();
034    // The model. Null if the model is not successfully loaded.
035    private AstModel model;
036    // The scope. Null if the model is not successfully loaded.
037    private Scope scope;
038    // The solver which solved the previous instance.
039    // Null if first instance not yet solved.
040    private ClaferSolver solver;
041    // The last file successfully loaded.
042    private File modelFile;
043
044    public void init() throws ScriptException {
045        engine.put("rsc", this);
046        // It's important to set the ScriptEngine.FILENAME since that's
047        // what will be printed in error messages.
048        engine.put(ScriptEngine.FILENAME, "solver.js");
049        // solver.js contains the mapping from Javascript to the methods
050        // in this class.
051        engine.eval(new InputStreamReader(JavascriptShell.class.getResourceAsStream("solver.js")));
052        engine.put(ScriptEngine.FILENAME, "command line");
053        engine.put("help",
054                "help             display this helpful message\n"
055                + "load()           reload the model\n"
056                + "load(filename)   load a new model\n"
057                + "solve()          find the next solution\n"
058                + "maximize(Clafer) find a solution where Clafer.ref is maximal\n"
059                + "minimize(Clafer) find a solution where Clafer.ref is minimal\n"
060                + "minUnsat()       find the smallest set of unsatisfiable constraints and a near-miss\n"
061                + "unsatCore()      find a small set of mutually unsatisfiable constraints\n"
062                + "exit()           stop the session");
063    }
064
065    /**
066     * Returns information about the final Choco CSP.
067     *
068     * @return information for debugging
069     */
070    public String internals() {
071        return internals(false);
072    }
073
074    /**
075     * Returns information about the final Choco CSP.
076     *
077     * @param initial debug initial state or current state
078     * @return information for debugging
079     */
080    public String internals(boolean initial) {
081        if (model == null) {
082            return "No model. Use \"load(filename)\" to load in a new model.";
083        }
084        if (initial) {
085            return ClaferCompiler.compile(model, scope).getInternalSolver().toString();
086        }
087        if (solver == null) {
088            return "Solve an instance first. Try \"solve()\".";
089        }
090        return solver.getInternalSolver().toString() + "\n" + solver.getInternalSolver().getMeasures();
091    }
092
093    /**
094     * Execute Javascript.
095     *
096     * @param script executable Javascript
097     * @return the value of the script, or null if it is a statement
098     * @throws ScriptException
099     */
100    public Object eval(String script) throws ScriptException {
101        return engine.eval(script);
102    }
103
104    /**
105     * Reload the previous successful file.
106     *
107     * @return a message
108     * @throws FileNotFoundException something happened to the file
109     * @throws ScriptException most likely a syntax error in the file
110     */
111    public String load() throws FileNotFoundException, ScriptException {
112        if (modelFile == null) {
113            return "No model. Use \"load(filename)\" to load in a new model.";
114        }
115        return load(modelFile);
116    }
117
118    /**
119     * Load a new file. The file must contain valid Javascript.
120     *
121     * @param filename the name of the file
122     * @return a message
123     * @throws FileNotFoundException something happened to the file
124     * @throws ScriptException most likely a syntax error in the file
125     */
126    public String load(String filename) throws FileNotFoundException, ScriptException {
127        return load(new File(filename));
128    }
129
130    /**
131     * Load a new file. The file must contain valid Javascript.
132     *
133     * @param file the file
134     * @return a message
135     * @throws FileNotFoundException something happened to the file
136     * @throws ScriptException most likely a syntax error in the file
137     */
138    public String load(File file) throws FileNotFoundException, ScriptException {
139        model = null;
140        scope = null;
141        solver = null;
142        Bindings bindings = engine.getBindings(ScriptContext.ENGINE_SCOPE);
143        try {
144            engine.setBindings(engine.createBindings(), ScriptContext.ENGINE_SCOPE);
145            Pair<AstModel, Scope> pair = Javascript.readModel(file, engine);
146            model = pair.getFst();
147            scope = pair.getSnd();
148        } catch (FileNotFoundException e) {
149            engine.setBindings(bindings, ScriptContext.ENGINE_SCOPE);
150            throw e;
151        } catch (ScriptException e) {
152            engine.setBindings(bindings, ScriptContext.ENGINE_SCOPE);
153            throw e;
154        } finally {
155            init();
156        }
157        modelFile = file;
158        return "Loaded " + AstUtil.getNames(AstUtil.getClafers(model)) + ".";
159    }
160
161    /**
162     * Find the first instance or next instances upon subsequent invocations.
163     *
164     * @return an instance
165     */
166    public InstanceModel solve() {
167        if (solver == null) {
168            if (model == null) {
169                throw new JavascriptException("No model. Use \"load(filename)\" to load in a new model.");
170            }
171            solver = ClaferCompiler.compile(model, scope);
172        }
173        return solver.find() ? solver.instance() : null;
174    }
175
176    /**
177     * Find an instance where the Clafer's value is maximal.
178     *
179     * @param clafer maximize this Clafer's value
180     * @return the maximal value and the optimal instance
181     */
182    public Pair<Integer, InstanceModel> maximize(AstClafer clafer) {
183        if (!clafer.hasRef()) {
184            throw new JavascriptException("Cannot maximize " + clafer + ".");
185        }
186        ClaferObjective objective = ClaferCompiler.compileMaximize(model, scope, clafer.getRef());
187        return objective.optimal();
188
189    }
190
191    /**
192     * Find an instance where the Clafer's value is minimal.
193     *
194     * @param clafer minimize this Clafer's value
195     * @return the minimal value and the optimal instance
196     */
197    public Pair<Integer, InstanceModel> minimize(AstClafer clafer) {
198        if (!clafer.hasRef()) {
199            throw new JavascriptException("Cannot minimize " + clafer + ".");
200        }
201        ClaferObjective objective = ClaferCompiler.compileMinimize(model, scope, clafer.getRef());
202        return objective.optimal();
203    }
204
205    /**
206     * Find the Min-Unsat and near-miss example. The Min-Unsat is empty if the
207     * model is satisfiable, in which case the near-miss example is a real
208     * instance of the model.
209     *
210     * @return the Min-Unsat and near-miss example
211     */
212    public Pair<Set<AstConstraint>, InstanceModel> minUnsat() {
213        ClaferUnsat unsat = ClaferCompiler.compileUnsat(model, scope);
214        return unsat.minUnsat();
215    }
216
217    /**
218     * Find the Min-Unsat-Core. Undefined if the model is satisfiable.
219     *
220     * @return the Min-Unsat-Core
221     */
222    public Set<AstConstraint> unsatCore() {
223        ClaferUnsat unsat = ClaferCompiler.compileUnsat(model, scope);
224        return unsat.unsatCore();
225    }

227    /**
228     * Exit the program.
229     *
230     * @return a message
231     */
232    public String exit() {
233        // Use interrupts to signal end.
234        Thread.currentThread().interrupt();
235        return "Exitting ...";
236    }

238    public static void main(String[] args) throws IOException, ScriptException {
239        JavascriptShell context = new JavascriptShell();
240        if (args.length > 0) {
241            try {
242                System.out.println(context.load(args[0]));
243            } catch (IOException e) {
244                System.out.println("Error: " + getOriginalMessage(e));
245            } catch (ScriptException e) {
246                System.out.println("Error: " + getOriginalMessage(e));
247            }
248        } else {
249            context.init();
250        }
251
252        System.out.println("Type \"help\" for a list of commands.");
253        System.out.print("> ");
254        System.out.flush();
255        BufferedReader r = new BufferedReader(new InputStreamReader(System.in));
256        String line;
257        while ((line = r.readLine()) != null) {
258            line = line.trim();
259            // Do not evaluate empty lines.
260            if (line.length() > 0) {
261                try {
262                    // Evaluate the Javascript command.
263                    System.out.println(context.eval(line));
264                    if (Thread.interrupted()) {
265                        // Exit was invoked.
266                        break;
267                    }
268                } catch (ScriptException e) {
269                    System.out.println("Error: " + getOriginalMessage(e));
270                }
271            }
272            System.out.print("> ");
273            System.out.flush();
274        }
275        System.out.println();
276    }
277
278    /**
279     * Find the message in the innermost cause. The Rhino engine tends to have
280     * several layers of wrapped exceptions, which do not look great when
281     * printing on the command line.
282     *
283     * @param e the outermost exception
284     * @return the innermost message
285     */
286    private static String getOriginalMessage(Throwable e) {
287        if (e.getCause() != null) {
288            return getOriginalMessage(e.getCause());
289        }
290        return e.getMessage();
291    }
