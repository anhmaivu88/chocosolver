003import java.util.Arrays;
004import solver.constraints.Propagator;

011/**
012 * strings[i] &lt; strings[j] iff ints[i] &lt;= ints[j] strings[i] = strings[j]
013 * iff ints[i] = ints[j]
014 *
015 * 0 &lt; ints[i] &lt; |{ints}|
016 *
017 * @author jimmy
018 */
019public class PropLexChainChannel extends Propagator<IntVar> {
020
021    private final IntVar[][] strings;
022    private final IntVar[] ints;
023
024    public PropLexChainChannel(IntVar[][] strings, IntVar[] ints) {
025        super(buildArray(strings, ints), PropagatorPriority.QUADRATIC, false);
026        this.strings = strings;
027        this.ints = ints;
028    }
029
030    public static IntVar[] buildArray(IntVar[][] strings, IntVar[] ints) {
031        if (strings.length != ints.length) {
032            throw new IllegalArgumentException();
033        }
034        IntVar[] array = new IntVar[strings.length * strings[0].length + ints.length];
035        System.arraycopy(ints, 0, array, 0, ints.length);
036        int i = ints.length;
037        for (IntVar[] string : strings) {
038            System.arraycopy(string, 0, array, i, string.length);
039            i += string.length;
040        }
041        assert i == array.length;
042        return array;
043    }
044
045    @Override
046    public int getPropagationConditions(int vIdx) {
047        return EventType.INT_ALL_MASK();
048    }
049
050    private static Ordering compareString(IntVar[] a, IntVar[] b) {
051        return compareString(a, b, 0);
052    }
053
054    private static Ordering compareString(IntVar[] a, IntVar[] b, int index) {
055        if (index == a.length) {
056            return a.length == b.length ? Ordering.EQ : Ordering.LT;
057        }
058        if (index == b.length) {
059            assert a.length != b.length;
060            return Ordering.GT;
061        }
062        Ordering ord = compare(a[index], b[index]);
063        switch (ord) {
064            case EQ:
065                return compareString(a, b, index + 1);
066            case LE:
067                switch (compareString(a, b, index + 1)) {
068                    case LT:
069                        return Ordering.LT;
070                    case LE:
071                    case EQ:
072                        return Ordering.LE;
073                    default:
074                        return Ordering.UNKNOWN;
075                }
076            case GE:
077                switch (compareString(a, b, index + 1)) {
078                    case GT:
079                        return Ordering.GT;
080                    case GE:
081                    case EQ:
082                        return Ordering.GE;
083                    default:
084                        return Ordering.UNKNOWN;
085                }
086            default:
087                return ord;
088        }
089    }
090
091    private static Ordering compare(IntVar a, IntVar b) {
092        if (a.instantiated() && b.instantiated() && a.getValue() == b.getValue()) {
093            return Ordering.EQ;
094        }
095        int aLb = a.getLB();
096        int aUb = a.getUB();
097        int bLb = b.getLB();
098        int bUb = b.getUB();
099        if (aLb > bUb) {
100            return Ordering.GT;
101        }
102        if (aLb >= bUb) {
103            return Ordering.GE;
104        }
105        if (aUb < bLb) {
106            return Ordering.LT;
107        }
108        if (aUb <= bLb) {
109            return Ordering.LE;
110        }
111        return Ordering.UNKNOWN;
112    }
113
114    private boolean lessThanString(IntVar[] a, IntVar[] b) throws ContradictionException {
115        boolean changed = false;
116        for (int x = 0; x < a.length - 1; x++) {
117            switch (compare(a[x], b[x])) {
118                case LT:
119                case LE:
120                case UNKNOWN:
121                    return changed;
122                case GT:
123                case GE:
124                    changed |= equal(a[x], b[x]);
125                    break;
126            }
127        }
128        // All equal except for the last character.
129        return lessThan(a[a.length - 1], b[a.length - 1]) || changed;
130    }
131
132    private boolean lessThan(IntVar a, IntVar b) throws ContradictionException {
133        // Don't use short-circuit or
134        return a.updateUpperBound(b.getUB() - 1, aCause)
135                | b.updateLowerBound(a.getLB() + 1, aCause);
136    }
137
138    private boolean lessThanEqualString(IntVar[] a, IntVar[] b) throws ContradictionException {
139        for (int x = 0; x < a.length; x++) {
140            Ordering charOrd = compare(a[x], b[x]);
141            if (!Ordering.EQ.equals(charOrd)) {
142                return lessThanEqual(a[x], b[x]);
143            }
144        }
145        // The two strings are equal.
146        return false;
147    }
148
149    private boolean lessThanEqual(IntVar a, IntVar b) throws ContradictionException {
150        // Don't use short-circuit or
151        return a.updateUpperBound(b.getUB(), aCause)
152                | b.updateLowerBound(a.getLB(), aCause);
153    }
154
155    private boolean greaterThanString(IntVar[] a, IntVar[] b) throws ContradictionException {
156        boolean changed = false;
157        for (int x = 0; x < a.length - 1; x++) {
158            switch (compare(a[x], b[x])) {
159                case GT:
160                case GE:
161                case UNKNOWN:
162                    return changed;
163                case LT:
164                case LE:
165                    changed |= equal(a[x], b[x]);
166                    break;
167            }
168        }
169        // All equal except for the last character.
170        return greaterThan(a[a.length - 1], b[a.length - 1]) || changed;
171    }
172
173    private boolean greaterThan(IntVar a, IntVar b) throws ContradictionException {
174        // Don't use short-circuit or
175        return a.updateLowerBound(b.getLB() + 1, aCause)
176                | b.updateUpperBound(a.getUB() - 1, aCause);
177    }
178
179    private boolean greaterThanEqualString(IntVar[] a, IntVar[] b) throws ContradictionException {
180        for (int x = 0; x < a.length; x++) {
181            Ordering charOrd = compare(a[x], b[x]);
182            if (!Ordering.EQ.equals(charOrd)) {
183                return greaterThanEqual(a[x], b[x]);
184            }
185        }
186        // The two strings are equal.
187        return false;
188    }
189
190    private boolean greaterThanEqual(IntVar a, IntVar b) throws ContradictionException {
191        // Don't use short-circuit or
192        return a.updateLowerBound(b.getLB(), aCause)
193                | b.updateUpperBound(a.getUB(), aCause);
194    }
195
196    private boolean equalString(IntVar[] a, IntVar[] b) throws ContradictionException {
197        boolean changed = false;
198        for (int i = 0; i < a.length; i++) {
199            changed |= equal(a[i], b[i]);
200        }
201        return changed;
202    }
203
204    private boolean equal(IntVar a, IntVar b) throws ContradictionException {
205        // Don't use short-circuit or
206        return a.updateLowerBound(b.getLB(), aCause)
207                | a.updateUpperBound(b.getUB(), aCause)
208                | b.updateLowerBound(a.getLB(), aCause)
209                | b.updateUpperBound(a.getUB(), aCause);
210    }
211
212    // Idempotent.
213    private boolean propagateStrings() throws ContradictionException {
214        Update operations = new Update(strings.length);
215        int eqs = 0;
216        for (int i = 0; i < strings.length; i++) {
217            boolean equivalenceClass = false;
218            for (int j = i + 1; j < strings.length; j++) {
219                Ordering ord = compareString(strings[i], strings[j]);
220                switch (ord) {
221                    case EQ:
222                        equivalenceClass = true;
223                    // fallthrough
224                    case LT:
225                    case LE:
226                    case GT:
227                    case GE:
228                        operations.add(i, j, ord);
229                        break;
230                }
231            }
232            if (equivalenceClass) {
233                eqs++;
234            }
235        }
236        boolean changed = false;
237        for (int i = 0; i < ints.length; i++) {
238            changed |= ints[i].updateUpperBound(ints.length - 1 - eqs, aCause);
239        }
240        boolean repeat;
241        do {
242            repeat = operations.doIntUpdates();
243            changed |= repeat;
244        } while (repeat);
245        return changed;
246    }
247
248    // Idempotent.
249    private boolean propagateInts() throws ContradictionException {
250        Update operations = new Update(strings.length);
251        for (int i = 0; i < ints.length; i++) {
252            for (int j = i + 1; j < ints.length; j++) {
253                Ordering ord = compare(ints[i], ints[j]);
254                if (!Ordering.UNKNOWN.equals(ord)) {
255                    operations.add(i, j, ord);
256                }
257            }
258        }
259        boolean changed = false;
260        boolean repeat;
261        do {
262            repeat = operations.doStringUpdates();
263            changed |= repeat;
264        } while (repeat);
265        return changed;
266    }
267
268    @Override
269    public void propagate(int evtmask) throws ContradictionException {
270        propagateStrings();
271        while (propagateInts() && propagateStrings());
272    }
273
274    @Override
275    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
276        forcePropagate(EventType.FULL_PROPAGATION);
277    }
278
279    @Override
280    public ESat isEntailed() {
281        for (int i = 0; i < strings.length; i++) {
282            for (int j = i + 1; j < strings.length; j++) {
283                Ordering intOrdering = compare(ints[i], ints[j]);
284                Ordering ord = Ordering.EQ;
285                for (int k = 0; Ordering.EQ.equals(ord) && k < strings[i].length; k++) {
286                    ord = compare(strings[i][k], strings[j][k]);
287                }
288                if (!intOrdering.equals(ord)
289                        && !Ordering.UNKNOWN.equals(intOrdering)
290                        && !Ordering.UNKNOWN.equals(ord)) {
291                    return ESat.FALSE;
292                }
293            }
294        }
295        return isCompletelyInstantiated() ? ESat.TRUE : ESat.UNDEFINED;
296    }
297
298    @Override
299    public String toString() {
300        return "lexChainChannel(" + Arrays.deepToString(strings) + ", " + Arrays.toString(ints) + ")";
301    }
302
303    private static enum Ordering {
304
305        LT,
306        LE,
307        GT,
308        GE,
309        EQ,
310        UNKNOWN;
311    }
312
313    private class Update {
314
315        private int[] updates;
316        private int size = 0;
317
318        Update(int capacity) {
319            updates = new int[capacity * 3];
320        }
321
322        void add(int x, int y, Ordering ord) {
323            int offset = size * 3;
324            if (offset >= updates.length) {
325                updates = Arrays.copyOf(updates, updates.length * 2);
326            }
327            updates[offset] = x;
328            updates[offset + 1] = y;
329            updates[offset + 2] = ord.ordinal();
330            size++;
331        }
332
333        boolean doStringUpdates() throws ContradictionException {
334            boolean changed = false;
335            int cap = size * 3;
336            for (int i = 0; i < cap; i += 3) {
337                int x = updates[i];
338                int y = updates[i + 1];
339                switch (updates[i + 2]) {
340                    case 0: // LT
341                        changed |= lessThanString(strings[x], strings[y]);
342                        break;
343                    case 1: // LE
344                        changed |= lessThanEqualString(strings[x], strings[y]);
345                        break;
346                    case 2: // GT
347                        changed |= greaterThanString(strings[x], strings[y]);
348                        break;
349                    case 3: // GE
350                        changed |= greaterThanEqualString(strings[x], strings[y]);
351                        break;
352                    case 4: // EQ
353                        changed |= equalString(strings[x], strings[y]);
354                        break;
355                    default:
356                        throw new IllegalStateException();
357                }
358            }
359            return changed;
360        }
361
362        boolean doIntUpdates() throws ContradictionException {
363            boolean changed = false;
364            int cap = size * 3;
365            for (int i = 0; i < cap; i += 3) {
366                int x = updates[i];
367                int y = updates[i + 1];
368                switch (updates[i + 2]) {
369                    case 0: // LT
370                        changed |= lessThan(ints[x], ints[y]);
371                        break;
372                    case 1: // LE
373                        changed |= lessThanEqual(ints[x], ints[y]);
374                        break;
375                    case 2: // GT
376                        changed |= greaterThan(ints[x], ints[y]);
377                        break;
378                    case 3: // GE
379                        changed |= greaterThanEqual(ints[x], ints[y]);
380                        break;
381                    case 4: // EQ
382                        changed |= equal(ints[x], ints[y]);
383                        break;
384                    default:
385                        throw new IllegalStateException();
386                }
387            }
388            return changed;
389        }
390    }
