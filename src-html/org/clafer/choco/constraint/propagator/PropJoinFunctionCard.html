019public class PropJoinFunctionCard extends Propagator<Variable> {

021    private final SetVar take;
022    private final IntVar takeCard;
023    private final IntVar[] refs;
024    private final IntVar toCard;
025    private final Integer globalCardinality;
026
027    public PropJoinFunctionCard(SetVar take, IntVar takeCard, IntVar[] refs, IntVar toCard, Integer globalCardinality) {
028        super(buildArray(take, takeCard, toCard, refs), PropagatorPriority.LINEAR, false);
029        this.take = take;
030        this.takeCard = takeCard;
031        this.refs = refs;
032        this.toCard = toCard;
033        this.globalCardinality = globalCardinality;
034    }
035
036    private static Variable[] buildArray(SetVar take, IntVar takeCard, IntVar toCard, IntVar[] refs) {
037        Variable[] array = new Variable[refs.length + 3];
038        array[0] = take;
039        array[1] = takeCard;
040        array[2] = toCard;
041        System.arraycopy(refs, 0, array, 3, refs.length);
042        return array;
043    }
044
045    private boolean isTakeVar(int idx) {
046        return idx == 0;
047    }
048
049    private boolean isTakeCardVar(int idx) {
050        return idx == 1;
051    }
052
053    private boolean isToCardVar(int idx) {
054        return idx == 2;
055    }
056
057    private boolean isRefVar(int idx) {
058        return idx >= 3;
059    }
060
061    private int getRefVarIndex(int idx) {
062        assert isRefVar(idx);
063        return idx - 3;
064    }
065
066    public boolean hasGlobalCardinality() {
067        return globalCardinality != null;
068    }
069
070    public int getGlobalCardinality() {
071        assert hasGlobalCardinality();
072        return globalCardinality.intValue();
073    }
074
075    @Override
076    public int getPropagationConditions(int vIdx) {
077        if (isTakeVar(vIdx)) {
078            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
079        }
080        if (isRefVar(vIdx)) {
081            return EventType.INSTANTIATE.mask;
082        }
083        assert isTakeCardVar(vIdx) || isToCardVar(vIdx);
084        return EventType.BOUND.mask + EventType.INSTANTIATE.mask;
085    }
086
087    private int countAdditionalSameRefsAllowed(TIntIntHashMap map) {
088        assert hasGlobalCardinality();
089        int gc = getGlobalCardinality();
090
091        int allowed = 0;
092
093        if (gc != 1) {
094            TIntIntIterator iter = map.iterator();
095            for (int i = map.size(); i-- > 0;) {
096                iter.advance();
097                assert iter.value() > 0;
098                assert iter.value() <= gc;
099                allowed += gc - iter.value();
100            }
101            assert !iter.hasNext();
102        }
103
104        return allowed;
105    }
106
107    private TIntIntHashMap countRefs() {
108        TIntIntHashMap map = new TIntIntHashMap(take.getKernelSize());
109        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
110            IntVar ref = refs[i];
111            if (ref.instantiated()) {
112                map.adjustOrPutValue(ref.getValue(), 1, 1);
113            }
114        }
115        return map;
116    }
117
118    private TIntIntHashMap constrainGlobalCardinality() throws ContradictionException {
119        assert hasGlobalCardinality();
120        int[] ker = PropUtil.iterateKer(take);
121        TIntIntHashMap map = new TIntIntHashMap(ker.length);
122        for (int i = 0; i < ker.length; i++) {
123            constrainGlobalCardinality(ker, i, i, map);
124        }
125        return map;
126    }
127
128    private void constrainGlobalCardinality(int[] ker, int index, int explored, TIntIntHashMap map) throws ContradictionException {
129        assert hasGlobalCardinality();
130        assert index <= explored;
131        assert explored < ker.length;
132
133        IntVar a = refs[ker[index]];
134        if (a.instantiated()) {
135            int value = a.getValue();
136            int count = map.adjustOrPutValue(value, 1, 1);
137            int gc = getGlobalCardinality();
138
139            if (count == gc) {
140                for (int j = 0; j < explored; j++) {
141                    IntVar b = refs[ker[j]];
142                    if (!b.instantiatedTo(value) && b.removeValue(value, aCause)) {
143                        constrainGlobalCardinality(ker, j, explored, map);
144                    }
145                }
146                for (int j = explored + 1; j < ker.length; j++) {
147                    refs[ker[j]].removeValue(value, aCause);
148                }
149            } else if (count > gc) {
150                contradiction(a, "Above global cardinality");
151            }
152        }
153    }
154
155    private static int divRoundUp(int a, int b) {
156        assert a >= 0;
157        assert b > 0;
158
159        return (a + b - 1) / b;
160    }
161
162    @Override
163    public void propagate(int evtmask) throws ContradictionException {
164        takeCard.updateLowerBound(take.getKernelSize(), aCause);
165        takeCard.updateUpperBound(take.getEnvelopeSize(), aCause);
166        boolean changed;
167        do {
168            changed = false;
169            TIntIntHashMap map = hasGlobalCardinality() ? constrainGlobalCardinality() : countRefs();
170            int instCard = map.size();
171            int kerSize = take.getKernelSize();
172
173            int minUninstantiated;
174            int maxUninstantiated;
175            int minCard;
176            int maxCard;
177            boolean cardChanged;
178            do {
179                cardChanged = false;
180//                minUninstantiated = Math.max(0, takeCard.getLB() - kerSize);
181//                maxUninstantiated = Math.max(0, takeCard.getUB() - kerSize);
182                int kerUninstantiated = 0;
183                for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
184                    if (!refs[i].instantiated()) {
185                        if (take.kernelContains(i)) {
186                            kerUninstantiated++;
187                        }
188                    }
189                }
190                assert takeCard.getLB() >= kerSize;
191                minUninstantiated = takeCard.getLB() - kerSize + kerUninstantiated;
192                maxUninstantiated = takeCard.getUB() - kerSize + kerUninstantiated;
193                minCard = instCard
194                        + (hasGlobalCardinality()
195                        ? divRoundUp(Math.max(0, minUninstantiated - countAdditionalSameRefsAllowed(map)), getGlobalCardinality())
196                        : 0);
197                maxCard = instCard + maxUninstantiated;
198
199                toCard.updateLowerBound(minCard, aCause);
200                toCard.updateUpperBound(maxCard, aCause);
201
202                if (hasGlobalCardinality()) {
203                    cardChanged |= takeCard.updateUpperBound(toCard.getUB() * getGlobalCardinality(), aCause);
204                }
205                cardChanged |= takeCard.updateLowerBound(toCard.getLB(), aCause);
206            } while (cardChanged);
207
208
209            if (maxUninstantiated != 0) {
210                if (instCard == toCard.getUB()) {
211                    // The rest must be duplicates.
212                    for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
213                        IntVar ref = refs[i];
214                        assert !ref.instantiated() || map.contains(ref.getValue());
215                        if (!ref.instantiated()) {
216                            changed |= PropUtil.domainSubsetOf(ref, map.keySet(), aCause) && ref.instantiated();
217                        }
218                    }
219                }
220                if (maxCard == toCard.getLB()) {
221                    // No more duplicate values.
222                    for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
223                        IntVar ref = refs[i];
224                        if (!ref.instantiated()) {
225                            TIntIntIterator iter = map.iterator();
226                            for (int j = map.size(); j-- > 0;) {
227                                iter.advance();
228                                changed |= ref.removeValue(iter.key(), aCause) && ref.instantiated();
229                            }
230                            assert !iter.hasNext();
231                        }
232                    }
233                }
234            }
235        } while (changed);
236    }

238    @Override
239    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
240        forcePropagate(EventType.FULL_PROPAGATION);
241    }

243    @Override
244    public ESat isEntailed() {
245        TIntIntHashMap map = new TIntIntHashMap();
246        int gc = hasGlobalCardinality() ? getGlobalCardinality() : Integer.MAX_VALUE;
247        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
248            IntVar ref = refs[i];
249            if (ref.instantiated()) {
250                if (map.adjustOrPutValue(ref.getValue(), 1, 1) > gc) {
251                    return ESat.FALSE;
252                }
253            }
254        }
255
256        int instCard = map.size();
257
258        int uninstantiated = 0;
259        map.clear();
260        int minUninstantiated = Math.max(0, takeCard.getLB() - take.getKernelSize());
261        int maxUninstantiated = Math.max(0, takeCard.getUB() - take.getKernelSize());
262        for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
263            if (!refs[i].instantiated()) {
264                if (take.kernelContains(i)) {
265                    minUninstantiated++;
266                }
267                maxUninstantiated++;
268            }
269        }
270        int minCard = instCard
271                + (hasGlobalCardinality()
272                ? divRoundUp(Math.max(0, minUninstantiated - countAdditionalSameRefsAllowed(map)), getGlobalCardinality())
273                : 0);
274        int maxCard = instCard + maxUninstantiated;
275
276        if (toCard.getUB() < minCard) {
277            return ESat.FALSE;
278        }
279        if (toCard.getLB() > maxCard) {
280            return ESat.FALSE;
281        }
282
283        return uninstantiated == 0 ? ESat.TRUE : ESat.UNDEFINED;
284    }

286    @Override
287    public String toString() {
288        return "joinFunctionCard(" + take + ", " + takeCard + ", " + Arrays.toString(refs) + ", " + toCard + ", " + globalCardinality + ")";
289    }
