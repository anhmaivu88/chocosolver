065    public static Constraint arrayToSet(IntVar[] ivars, SetVar svar, IntVar svarCard, Integer globalCardinality) {
066        Variable[] vars = new Variable[ivars.length + 2];
067        vars[0] = svar;
068        vars[1] = svarCard;
069        System.arraycopy(ivars, 0, vars, 2, ivars.length);
070        Constraint constraint = new Constraint(vars, svar.getSolver());
071        constraint.setPropagators(
072                new PropArrayToSet(ivars, svar),
073                new PropArrayToSetCard(ivars, svarCard, globalCardinality),
074                new PropCardinality(svar, svarCard));
075        return constraint;
076    }

110    public static Constraint joinRelation(SetVar take, SetVar[] children, SetVar to) {
111        SetVar[] vars = new SetVar[children.length + 2];
112        vars[0] = take;
113        vars[1] = to;
114        System.arraycopy(children, 0, vars, 2, children.length);
115        Constraint<SetVar, PropJoinRelation> constraint = new Constraint<SetVar, PropJoinRelation>(vars, take.getSolver());
116        constraint.setPropagators(new PropJoinRelation(take, children, to));
117        return constraint;
118    }

120    public static Constraint joinInjectiveRelation(SetVar take, IntVar takeCard, SetVar[] children, IntVar[] childrenCards, SetVar to, IntVar toCard) {
121        if (children.length != childrenCards.length) {
122            throw new IllegalArgumentException();
123        }
124        Variable[] vars = new Variable[children.length * 2 + 4];
125        vars[0] = take;
126        vars[1] = takeCard;
127        vars[2] = to;
128        vars[3] = toCard;
129        System.arraycopy(children, 0, vars, 4, children.length);
130        System.arraycopy(childrenCards, 0, vars, 4 + children.length, childrenCards.length);
131        // Assumes take and children card is already constrained for maximum efficiency.
132        Constraint constraint = new Constraint(vars, take.getSolver());
133        constraint.setPropagators(
134                new PropJoinRelation(take, children, to),
135                new PropJoinInjectiveRelationCard(take, takeCard, childrenCards, toCard),
136                new PropCardinality(to, toCard));
137        return constraint;
138    }

144    public static Constraint joinFunction(SetVar take, IntVar takeCard, IntVar[] refs, SetVar to, IntVar toCard, Integer globalCardinality) {
145        Variable[] vars = new Variable[refs.length + 4];
146        vars[0] = take;
147        vars[1] = takeCard;
148        vars[2] = to;
149        vars[3] = toCard;
150        // Assumes take card is already constrained for maximum efficiency.
151        System.arraycopy(refs, 0, vars, 4, refs.length);
152        Constraint constraint = new Constraint(vars, take.getSolver());
153        constraint.setPropagators(new PropJoinFunction(take, refs, to),
154                new PropJoinFunctionCard(take, takeCard, refs, toCard, globalCardinality),
155                new PropCardinality(to, toCard));
156        return constraint;
157    }

177    /**
178     * <p>
179     * Sum the set with a maximum cardinality. More efficient than the standard
180     * operation in the Choco library when n is relatively small.
181     * </p>
182     * <p>
183     * For example:
184     * <pre>
185     *   Animal 2
186     *     Age -> integer
187     *   [Animal.Age = 1000]
188     * </pre>
189     * </p>
190     * <p> {@code Animal.Age} is a set with a very large envelope. However, due
191     * to static analysis of the model, it is easy to see that the cardinality
192     * can be no larger than 2, hence n=2. Once the first integer x is selected
193     * for the set, the second integer 1000 - x is already determined due to
194     * n=2. Since the Choco library's setSum constraint is not given n, it
195     * cannot make this deduction.
196     * </p>
197     *
198     * @param set the set of integers
199     * @param sum the sum of the set
200     * @param card the cardinality of the set
201     * @return a constraint where and sum = Σ set
202     */
203    public static Constraint setSum(SetVar set, IntVar sum, IntVar card) {
204        Constraint constraint = new Constraint(new Variable[]{set, sum, card}, set.getSolver());
205        constraint.setPropagators(new PropSetSum(set, sum, card));
206        return constraint;
207    }

253    public static Constraint union(SetVar[] sets, IntVar[] setCards, SetVar union, IntVar unionCard) {
254        Variable[] array = new Variable[sets.length + setCards.length + 2];
255        System.arraycopy(sets, 0, array, 0, sets.length);
256        System.arraycopy(setCards, 0, array, sets.length, setCards.length);
257        array[sets.length + setCards.length] = union;
258        array[sets.length + setCards.length + 1] = unionCard;
259        Constraint constraint = new Constraint(array, union.getSolver());
260        constraint.setPropagators(new PropSetUnion(sets, union),
261                new PropSetUnionCard(setCards, unionCard),
262                new PropCardinality(union, unionCard));
263        return constraint;
264    }
