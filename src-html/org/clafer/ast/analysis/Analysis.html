097    private static Map<AstConcreteClafer, Card> buildCardMap(List<AstClafer> clafers) {
098        Map<AstConcreteClafer, Card> cardMap = new HashMap<AstConcreteClafer, Card>();
099        for (AstClafer clafer : clafers) {
100            if (clafer instanceof AstConcreteClafer) {
101                AstConcreteClafer concreteClafer = (AstConcreteClafer) clafer;
102                cardMap.put(concreteClafer, concreteClafer.getCard());
103            }
104        }
105        return cardMap;
106    }
107
108    private static List<AstClafer> append(List<? extends AstClafer> abstractClafers, List<? extends AstConcreteClafer> concreteClafers) {
109        List<AstClafer> clafers = new ArrayList<AstClafer>(abstractClafers.size() + concreteClafers.size());
110        clafers.addAll(abstractClafers);
111        clafers.addAll(concreteClafers);
112        return clafers;
113    }

115    public static Analysis analyze(AstModel model, Scope scope, Analyzer... analyzers) {
116        Analysis analysis = new Analysis(model, scope);
117        for (Analyzer analyzer : analyzers) {
118            analysis = analyzer.analyze(analysis);
119        }
120        return analysis;
121    }

123    private <T> T notNull(String analysisName, T t) {
124        if (t == null) {
125            throw new AnalysisException(analysisName + " not yet analyzed.");
126        }
127        return t;
128    }
129
130    private <T> T notNull(String key, String analysisName, T t) {
131        if (t == null) {
132            throw new AnalysisException(analysisName + " for " + key + " not yet analyzed.");
133        }
134        return t;
135    }
136
137    private <T> T notNull(AstClafer key, String analysisName, T t) {
138        return notNull(key.getName(), analysisName, t);
139    }

249    public Pair<AstConcreteClafer, Integer> getConcreteId(AstClafer clafer, int id) {
250        AstClafer sub = clafer;
251        int curId = id;
252        while (sub instanceof AstAbstractClafer) {
253            Offsets offset = getOffsets((AstAbstractClafer) sub);
254            sub = offset.getClafer(curId);
255            curId -= offset.getOffset(sub);
256        }
257        return new Pair<AstConcreteClafer, Integer>((AstConcreteClafer) sub, curId);
258    }
259
260    public Pair<AstClafer, Integer> getSubId(AstAbstractClafer clafer, int id) {
261        Offsets offsets = getOffsets(clafer);
262        AstClafer sub = offsets.getClafer(id);
263        return new Pair<AstClafer, Integer>(sub, id - offsets.getOffset(sub));
264    }

266    public Pair<AstAbstractClafer, Integer> getSuperId(AstClafer clafer, int id) {
267        assert clafer.hasSuperClafer();
268        int offset = getOffsets(clafer.getSuperClafer()).getOffset(clafer);
269        return new Pair<AstAbstractClafer, Integer>(clafer.getSuperClafer(), id + offset);
270    }
271
272    public List<Pair<AstAbstractClafer, Integer>> getSuperIds(AstClafer clafer, int id) {
273        List<Pair<AstAbstractClafer, Integer>> superIds = new ArrayList<Pair<AstAbstractClafer, Integer>>();
274        AstClafer sup = clafer;
275        int curId = id;
276        while (sup.hasSuperClafer()) {
277            Pair<AstAbstractClafer, Integer> superId = getSuperId(sup, curId);
278            superIds.add(superId);
279            sup = superId.getFst();
280            curId = superId.getSnd().intValue();
281        }
282        return superIds;
283    }

289    public List<Pair<AstClafer, Integer>> getHierarcyIds(AstClafer clafer, int id) {
290        List<Pair<AstClafer, Integer>> superIds = new ArrayList<Pair<AstClafer, Integer>>();
291        superIds.add(new Pair<AstClafer, Integer>(clafer, id));
292        AstClafer sup = clafer;
293        int curId = id;
294        while (sup.hasSuperClafer()) {
295            Pair<AstAbstractClafer, Integer> superId = getSuperId(sup, curId);
296            superIds.add(new Pair<AstClafer, Integer>(superId));
297            sup = superId.getFst();
298            curId = superId.getSnd().intValue();
299        }
300        return superIds;
301    }

303    public Pair<AstRef, Integer> getInheritedRefId(AstClafer clafer) {
304        AstClafer sup = clafer;
305        int curId = 0;
306        do {
307            if (sup.hasRef()) {
308                return new Pair<AstRef, Integer>(sup.getRef(), curId);
309            }
310            if (sup.hasSuperClafer()) {
311                curId += getOffsets(sup.getSuperClafer()).getOffset(sup);
312            }
313            sup = sup.getSuperClafer();
314        } while (sup != null);
315        return null;
316    }

357    public boolean hasInteritedBreakableChildren(AstClafer clafer) {
358        AstClafer sup = clafer;
359        do {
360            if (hasBreakableChildren(sup)) {
361                return true;
362            }
363            sup = sup.getSuperClafer();
364        } while (sup != null);
365        return false;
366    }

389    public boolean isBreakableRefId(AstRef ref, int id) {
390        int[] breakbleIDs = getBreakableRefsMap().get(ref);
391        if (breakbleIDs == null) {
392            return false;
393        }
394        return Util.in(id, breakbleIDs);
395    }
