001package org.clafer.compiler;
002
003import org.clafer.common.Check;
004import org.clafer.collection.Pair;
005import org.clafer.instance.InstanceModel;
006import solver.ResolutionPolicy;
007import solver.Solver;
008import solver.search.measure.IMeasures;
009import solver.variables.IntVar;
010import util.ESat;
011
012/**
013 *
014 * @author jimmy
015 */
016public class ClaferObjective {
017
018    public final Solver solver;
019    private final ClaferSolutionMap solutionMap;
020    private final Objective objective;
021    private final IntVar score;
022
023    ClaferObjective(Solver solver, ClaferSolutionMap solutionMap, Objective objective, IntVar score) {
024        this.solver = Check.notNull(solver);
025        this.solutionMap = Check.notNull(solutionMap);
026        this.objective = Check.notNull(objective);
027        this.score = Check.notNull(score);
028    }
029
030    public Solver getInternalSolver() {
031        return solver;
032    }
033
034    public IMeasures getMeasures() {
035        return solver.getMeasures();
036    }
037
038    public Objective getObjective() {
039        return objective;
040    }
041
042    public Pair<Integer, InstanceModel> optimal() {
043        solver.findOptimalSolution(objective.getPolicy(), score);
044        return ESat.TRUE.equals(solver.isFeasible())
045                ? new Pair<Integer, InstanceModel>(score.getValue(), solutionMap.getInstance())
046                : null;
047    }
048
049    public static enum Objective {
050
051        Maximize(ResolutionPolicy.MAXIMIZE),
052        Minimize(ResolutionPolicy.MINIMIZE);
053        private final ResolutionPolicy policy;
054
055        Objective(ResolutionPolicy policy) {
056            this.policy = policy;
057        }
058
059        ResolutionPolicy getPolicy() {
060            return policy;
061        }
062    }
063}
