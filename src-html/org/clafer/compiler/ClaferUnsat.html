001package org.clafer.compiler;
002
003import java.util.ArrayList;
004import java.util.HashSet;
005import java.util.List;
006import java.util.Set;
007import org.clafer.ast.AstConstraint;
008import org.clafer.common.Check;
009import org.clafer.collection.Pair;
010import org.clafer.instance.InstanceModel;
011import solver.ResolutionPolicy;
012import solver.Solver;
013import solver.constraints.ICF;
014import solver.variables.BoolVar;
015import solver.variables.IntVar;
016import util.ESat;
017
018/**
019 * Either call {@link #minUnsat()} xor {@link #unsatCore()} at most once. If you
020 * need to invoke both, you need to two ClaferUnsat objects.
021 *
022 * @author jimmy
023 */
024public class ClaferUnsat {
025
026    private final Solver solver;
027    private final ClaferSolutionMap solutionMap;
028    private final Pair<AstConstraint, BoolVar>[] softVars;
029    private final IntVar score;
030
031    ClaferUnsat(Solver solver, ClaferSolutionMap solutionMap, Pair<AstConstraint, BoolVar>[] softVars, IntVar score) {
032        this.solver = Check.notNull(solver);
033        this.solutionMap = Check.notNull(solutionMap);
034        this.softVars = Check.noNulls(softVars);
035        this.score = Check.notNull(score);
036    }
037
038    public Solver getInternalSolver() {
039        return solver;
040    }
041
042    /**
043     * Compute the minimal set of constraints that need to be removed before the
044     * model is satisfiable. If the model is already satisfiable, then the set
045     * is empty. Guaranteed to be minimum.
046     *
047     * @return the Min-Unsat and the corresponding near-miss example
048     */
049    public Pair<Set<AstConstraint>, InstanceModel> minUnsat() {
050        if (ESat.TRUE.equals(maximize())) {
051            Set<AstConstraint> unsat = new HashSet<AstConstraint>();
052            for (Pair<AstConstraint, BoolVar> softVar : softVars) {
053                if (softVar.getSnd().instantiatedTo(0)) {
054                    unsat.add(softVar.getFst());
055                }
056            }
057            return new Pair<Set<AstConstraint>, InstanceModel>(unsat, solutionMap.getInstance());
058        }
059        return null;
060    }
061
062    /**
063     * Compute a small set of constraints that are mutually unsatisfiable.
064     * Undefined behaviour if the model is satisfiable. This method is always
065     * slower to compute than {@link #minUnsat()}. Not guaranteed to be minimum.
066     *
067     * @return the Unsat-Core
068     */
069    public Set<AstConstraint> unsatCore() {
070        Set<AstConstraint> unsat = new HashSet<AstConstraint>();
071        boolean changed = true;
072        while (changed && ESat.TRUE.equals(maximize())) {
073            changed = false;
074            List<BoolVar> minUnsat = new ArrayList<BoolVar>();
075            for (Pair<AstConstraint, BoolVar> softVar : softVars) {
076                if (softVar.getSnd().instantiatedTo(0)) {
077                    changed |= unsat.add(softVar.getFst());
078                    minUnsat.add(softVar.getSnd());
079                }
080            }
081            solver.getSearchLoop().reset();
082            for (BoolVar var : minUnsat) {
083                solver.postCut(ICF.arithm(var, "=", 1));
084            }
085        }
086        return unsat;
087    }
088
089    private ESat maximize() {
090        solver.findOptimalSolution(ResolutionPolicy.MAXIMIZE, score);
091        return solver.isFeasible();
092    }
093}
