001package org.clafer.ir;
002
003import org.clafer.common.Check;
004
005/**
006 * left `op` right + offset
007 *
008 * @author jimmy
009 */
010public class IrCompare extends IrAbstractBool implements IrBoolExpr {
011
012    private final IrIntExpr left;
013    private final Op op;
014    private final IrIntExpr right;
015
016    IrCompare(IrIntExpr left, Op op, IrIntExpr right, IrBoolDomain domain) {
017        super(domain);
018        this.left = Check.notNull(left);
019        this.op = Check.notNull(op);
020        this.right = Check.notNull(right);
021    }
022
023    public IrIntExpr getLeft() {
024        return left;
025    }
026
027    public Op getOp() {
028        return op;
029    }
030
031    public IrIntExpr getRight() {
032        return right;
033    }
034
035    @Override
036    public IrBoolExpr negate() {
037        return new IrCompare(left, op.negate(), right, getDomain().invert());
038    }
039
040    @Override
041    public boolean isNegative() {
042        return false;
043    }
044
045    @Override
046    public <A, B> B accept(IrBoolExprVisitor<A, B> visitor, A a) {
047        return visitor.visit(this, a);
048    }
049
050    @Override
051    public <A, B> B accept(IrIntExprVisitor<A, B> visitor, A a) {
052        return visitor.visit(this, a);
053    }
054
055    @Override
056    public boolean equals(Object obj) {
057        if (obj instanceof IrCompare) {
058            IrCompare other = (IrCompare) obj;
059            return left.equals(other.left) && op.equals(other.op) && right.equals(other.right) && super.equals(other);
060        }
061        return false;
062    }
063
064    @Override
065    public int hashCode() {
066        // op.hashCode() can change between runs which makes the output change
067        // every time.
068        return left.hashCode() ^ op.ordinal() ^ right.hashCode();
069    }
070
071    @Override
072    public String toString() {
073        return left + " " + op.getSyntax() + " " + right;
074    }
075
076    public static enum Op {
077
078        Equal("="),
079        NotEqual("!="),
080        LessThan("<"),
081        LessThanEqual("<="),
082        GreaterThan(">"),
083        GreaterThanEqual(">=");
084        private final String syntax;
085
086        Op(String syntax) {
087            this.syntax = syntax;
088        }
089
090        public String getSyntax() {
091            return syntax;
092        }
093
094        public boolean isEquality() {
095            switch (this) {
096                case Equal:
097                case NotEqual:
098                    return true;
099                default:
100                    return false;
101            }
102        }
103
104        public boolean isInequality() {
105            return !isEquality();
106        }
107
108        public Op reverse() {
109            switch (this) {
110                case Equal:
111                    return Equal;
112                case NotEqual:
113                    return NotEqual;
114                case LessThan:
115                    return GreaterThan;
116                case LessThanEqual:
117                    return GreaterThanEqual;
118                case GreaterThan:
119                    return LessThan;
120                case GreaterThanEqual:
121                    return LessThanEqual;
122                default:
123                    throw new IllegalStateException();
124            }
125        }
126
127        public Op negate() {
128            switch (this) {
129                case Equal:
130                    return NotEqual;
131                case NotEqual:
132                    return Equal;
133                case LessThan:
134                    return GreaterThanEqual;
135                case LessThanEqual:
136                    return GreaterThan;
137                case GreaterThan:
138                    return LessThanEqual;
139                case GreaterThanEqual:
140                    return LessThan;
141                default:
142                    throw new IllegalStateException();
143            }
144        }
145    }
146}
