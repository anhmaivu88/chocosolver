016    private static <T> boolean changed(T t1, T t2) {
017        return t1 != t2;
018    }
019
020    private static <T> boolean changed(T[] t1, T[] t2) {
021        if (t1 == t2) {
022            return false;
023        }
024        if (t1.length != t2.length) {
025            return true;
026        }
027        for (int i = 0; i < t1.length; i++) {
028            if (changed(t1[i], t2[i])) {
029                return true;
030            }
031        }
032        return false;
033    }
034
035    private static <T> boolean changed(T[][] t1, T[][] t2) {
036        if (t1 == t2) {
037            return false;
038        }
039        if (t1.length != t2.length) {
040            return true;
041        }
042        for (int i = 0; i < t1.length; i++) {
043            if (changed(t1[i], t2[i])) {
044                return true;
045            }
046        }
047        return false;
048    }

050    public IrModule rewrite(IrModule module, T t) {
051        List<IrBoolExpr> rewrittenConstraints = new ArrayList<IrBoolExpr>(module.getConstraints().size());
052        for (IrBoolExpr constraint : module.getConstraints()) {
053            rewrittenConstraints.add(rewrite(constraint, t));
054        }
055        return new IrModule().addConstraints(rewrittenConstraints);
056    }
057
058    public IrModule rewriteAndNonNops(IrModule module, T t) {
059        Pair<List<IrNop>, List<IrBoolExpr>> pair = partitionNops(module.getConstraints());
060        IrModule rewritten = new IrModule().addConstraint(rewrite(and(pair.getSnd()), t));
061        rewritten.addConstraints(pair.getFst());
062        return rewritten;
063    }
064
065    private static Pair<List<IrNop>, List<IrBoolExpr>> partitionNops(List<IrBoolExpr> constraints) {
066        List<IrNop> nops = new ArrayList<IrNop>();
067        List<IrBoolExpr> nonNops = new ArrayList<IrBoolExpr>(constraints.size());
068
069        for (IrBoolExpr constraint : constraints) {
070            if (constraint instanceof IrNop) {
071                nops.add((IrNop) constraint);
072            } else {
073                nonNops.add(constraint);
074            }
075        }
076        return new Pair<List<IrNop>, List<IrBoolExpr>>(nops, nonNops);
077    }

079    public IrBoolExpr rewrite(IrBoolExpr expr, T t) {

083    public IrBoolExpr[] rewrite(IrBoolExpr[] exprs, T t) {
084        IrBoolExpr[] rewritten = new IrBoolExpr[exprs.length];
085        for (int i = 0; i < rewritten.length; i++) {
086            rewritten[i] = rewrite(exprs[i], t);
087        }
088        return rewritten;
089    }

178        IrBoolExpr antecedent = rewrite(ir.getAntecedent(), a);
179        IrBoolExpr consequent = rewrite(ir.getConsequent(), a);
180        return changed(ir.getAntecedent(), antecedent)
181                || changed(ir.getConsequent(), consequent)
182                ? implies(antecedent, consequent)
183                : ir;
