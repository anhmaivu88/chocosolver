051    public static IrDomain boundDomain(int low, int high) {
052        if (low == 0 && high == 0) {
053            return ZeroDomain;
054        }
055        if (low == 1 && high == 1) {
056            return OneDomain;
057        }
058        if (low == 0 && high == 1) {
059            return ZeroOneDomain;
060        }
061        return new IrBoundDomain(low, high);
062    }

072    public static IrDomain enumDomain(TIntHashSet values) {
073        switch (values.size()) {
074            case 0:
075                return EmptyDomain;
076            case 1:
077                int value = values.iterator().next();
078                return boundDomain(value, value);
079            default:
080                int[] array = values.toArray();
081                Arrays.sort(array);
082                // If the values are over a contiguous interval, then return a bound domain.
083                int low = array[0];
084                for (int i = 1; i < array.length; i++) {
085                    if (low + i != array[i]) {
086                        return new IrEnumDomain(array);
087                    }
088                }
089                // A contigious interval.
090                return boundDomain(low, array[array.length - 1]);
091        }
092    }

124    public static IrBoolExpr and(IrBoolExpr... operands) {
125        List<IrBoolExpr> flatten = new ArrayList<IrBoolExpr>(operands.length);
126        for (IrBoolExpr operand : operands) {
127            flattenAnd(operand, flatten);
128        }
129        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(flatten.size());
130        for (IrBoolExpr operand : flatten) {
131            if (IrUtil.isFalse(operand)) {
132                return False;
133            }
134            if (!IrUtil.isTrue(operand)) {
135                filter.add(operand);
136            }
137        }
138        switch (filter.size()) {
139            case 0:
140                return True;
141            case 1:
142                return filter.get(0);
143            default:
144                return new IrAnd(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
145        }
146    }

148    private static void flattenAnd(IrBoolExpr expr, List<IrBoolExpr> flatten) {
149        if (expr instanceof IrAnd) {
150            IrAnd and = (IrAnd) expr;
151            for (IrBoolExpr subexpr : and.getOperands()) {
152                flattenAnd(subexpr, flatten);
153            }
154        } else {
155            flatten.add(expr);
156        }
157    }

163    public static IrBoolExpr lone(IrBoolExpr... operands) {
164        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(operands.length);
165        int count = 0;
166        for (IrBoolExpr operand : operands) {
167            if (IrUtil.isTrue(operand)) {
168                count++;
169                if (count > 1) {
170                    return False;
171                }
172            } else if (!IrUtil.isFalse(operand)) {
173                filter.add(operand);
174            }
175        }
176        assert count == 0 || count == 1;
177        switch (filter.size()) {
178            case 0:
179                return True;
180            case 1:
181                return count == 0 ? True : not(filter.get(0));
182            default:
183                IrBoolExpr[] f = filter.toArray(new IrBoolExpr[filter.size()]);
184                return count == 0
185                        ? new IrLone(f, BoolDomain)
186                        : not(or(f));
187        }
188    }

194    public static IrBoolExpr one(IrBoolExpr... operands) {
195        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(operands.length);
196        int count = 0;
197        for (IrBoolExpr operand : operands) {
198            if (IrUtil.isTrue(operand)) {
199                count++;
200                if (count > 1) {
201                    return False;
202                }
203            }
204            if (!IrUtil.isFalse(operand)) {
205                filter.add(operand);
206            }
207        }
208        assert count == 0 || count == 1;
209        switch (filter.size()) {
210            case 0:
211                return count == 0 ? False : True;
212            case 1:
213                return count == 0 ? filter.get(0) : not(filter.get(0));
214            case 2:
215                return count == 0
216                        ? xor(filter.get(0), filter.get(1))
217                        : and(not(filter.get(0)), not(filter.get(1)));
218            default:
219                IrBoolExpr[] f = filter.toArray(new IrBoolExpr[filter.size()]);
220                return count == 0
221                        ? new IrOne(f, BoolDomain)
222                        : not(or(f));
223        }
224    }

265    public static IrBoolExpr implies(IrBoolExpr antecedent, IrBoolExpr consequent) {
266        if (IrUtil.isTrue(antecedent)) {
267            return consequent;
268        }
269        if (IrUtil.isFalse(antecedent)) {
270            return True;
271        }
272        if (IrUtil.isTrue(consequent)) {
273            return True;
274        }
275        if (IrUtil.isFalse(consequent)) {
276            return not(antecedent);
277        }
278        return new IrImplies(antecedent, consequent, BoolDomain);
279    }

297    public static IrBoolExpr ifThenElse(IrBoolExpr antecedent, IrBoolExpr consequent, IrBoolExpr alternative) {
298        if (IrUtil.isTrue(antecedent)) {
299            return consequent;
300        }
301        if (IrUtil.isFalse(antecedent)) {
302            return alternative;
303        }
304        if (IrUtil.isTrue(consequent)) {
305            return or(antecedent, alternative);
306        }
307        if (IrUtil.isFalse(consequent)) {
308            return and(antecedent.negate(), alternative);
309        }
310        if (IrUtil.isTrue(alternative)) {
311            return or(antecedent.negate(), consequent);
312        }
313        if (IrUtil.isFalse(alternative)) {
314            return and(antecedent, consequent);
315        }
316        return new IrIfThenElse(antecedent, consequent, alternative, BoolDomain);
317    }

324            return right;

327            return not(right);
328        }

330            return left;

332        if (IrUtil.isFalse(right)) {
333            return not(left);
334        }
335        return new IrIfOnlyIf(left, right, BoolDomain);

355        IrDomain domain = value.getDomain();
356        if (range.isBounded()
357                && domain.getLowBound() >= range.getLowBound()
358                && domain.getHighBound() <= range.getHighBound()) {
359            return True;
360        }
361        if (domain.getLowBound() > range.getHighBound()
362                || domain.getHighBound() < range.getLowBound()) {
363            return False;
364        }
365        return new IrWithin(value, range, BoolDomain);
366    }

390    public static IrBoolExpr compare(IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
391        IrDomain leftDomain = left.getDomain();
392        IrDomain rightDomain = right.getDomain();
393        switch (op) {
394            case Equal:
395                if (left.equals(right)) {
396                    return True;
397                }
398                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
399                    return constant(leftDomain.getLowBound() == rightDomain.getLowBound());
400                }
401                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
402                    return ifOnlyIf((IrBoolExpr) left, (IrBoolExpr) right);
403                }
404                break;
405            case NotEqual:
406                if (left.equals(right)) {
407                    return False;
408                }
409                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
410                    return constant(leftDomain.getLowBound() != rightDomain.getLowBound());
411                }
412                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
413                    return xor((IrBoolExpr) left, (IrBoolExpr) right);
414                }
415                break;
416            case LessThan:
417                if (left.equals(right)) {
418                    return False;
419                }
420                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
421                    return True;
422                }
423                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
424                    return False;
425                }
426                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
427                    return not(implies((IrBoolExpr) right, (IrBoolExpr) left));
428                }
429                break;
430            case LessThanEqual:
431                if (left.equals(right)) {
432                    return True;
433                }
434                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
435                    return True;
436                }
437                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
438                    return False;
439                }
440                if (leftDomain.getLowBound() == rightDomain.getHighBound()) {
441                    return equal(left, right);
442                }
443                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
444                    return implies((IrBoolExpr) left, (IrBoolExpr) right);
445                }
446                break;
447            case GreaterThan:
448                if (left.equals(right)) {
449                    return False;
450                }
451                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
452                    return True;
453                }
454                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
455                    return False;
456                }
457                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
458                    return not(implies((IrBoolExpr) left, (IrBoolExpr) right));
459                }
460                break;
461            case GreaterThanEqual:
462                if (left.equals(right)) {
463                    return True;
464                }
465                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
466                    return True;
467                }
468                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
469                    return False;
470                }
471                if (leftDomain.getHighBound() == rightDomain.getLowBound()) {
472                    return equal(left, right);
473                }
474                if (left instanceof IrBoolExpr && right instanceof IrBoolExpr) {
475                    return implies((IrBoolExpr) right, (IrBoolExpr) left);
476                }
477                break;
478            default:
479                throw new IllegalArgumentException("Unknown op: " + op);
480        }
481        return new IrCompare(left, op, right, BoolDomain);
482    }

496    public static IrBoolExpr equality(IrSetExpr left, IrSetTest.Op op, IrSetExpr right) {
497        switch (op) {
498            case Equal:
499                if (left.equals(right)) {
500                    return True;
501                }
502                int[] constant = IrUtil.getConstant(left);
503                if (constant != null) {
504                    if (constant.length == 0) {
505                        return equal(card(right), 0);
506                    }
507                    if (constant.length == right.getEnv().size()) {
508                        if (IrUtil.containsAll(constant, right.getEnv())) {
509                            return equal(card(right), constant.length);
510                        }
511                    }
512                }
513                constant = IrUtil.getConstant(right);
514                if (constant != null) {
515                    if (constant.length == 0) {
516                        return equal(card(left), 0);
517                    }
518                    if (constant.length == left.getEnv().size()) {
519                        if (IrUtil.containsAll(constant, left.getEnv())) {
520                            return equal(card(left), constant.length);
521                        }
522                    }
523                }
524                break;
525            case NotEqual:
526                if (left.equals(right)) {
527                    return False;
528                }
529                constant = IrUtil.getConstant(left);
530                if (constant != null) {
531                    if (constant.length == 0) {
532                        return equal(card(right), 0);
533                    }
534                    if (constant.length == right.getEnv().size()) {
535                        if (IrUtil.containsAll(constant, right.getEnv())) {
536                            return notEqual(card(right), constant.length);
537                        }
538                    }
539                }
540                constant = IrUtil.getConstant(right);
541                if (constant != null) {
542                    if (constant.length == 0) {
543                        return equal(card(left), 0);
544                    }
545                    if (constant.length == left.getEnv().size()) {
546                        if (IrUtil.containsAll(constant, left.getEnv())) {
547                            return notEqual(card(left), constant.length);
548                        }
549                    }
550                }
551                break;
552            default:
553                throw new IllegalArgumentException();
554        }
555        return new IrSetTest(left, op, right, BoolDomain);
556    }

628            return True;

631            return False;

666    public static IrBoolExpr boolChannel(IrBoolExpr[] bools, IrSetExpr set) {
667        {
668            int[] constant = IrUtil.getConstant(set);
669            if (constant != null) {
670                IrBoolExpr[] ands = new IrBoolExpr[bools.length];
671                for (int i = 0; i < ands.length; i++) {
672                    ands[i] = equal(bools[i], Util.in(i, constant) ? True : False);
673                }
674                return and(ands);
675            }
676        }
677        TIntHashSet values = new TIntHashSet();
678        IrDomain env = set.getEnv();
679        IrDomain ker = set.getKer();
680        boolean entailed = true;
681        for (int i = 0; i < bools.length; i++) {
682            Boolean constant = IrUtil.getConstant(bools[i]);
683            if (Boolean.TRUE.equals(constant)) {
684                if (values != null) {
685                    values.add(i);
686                }
687                if (!env.contains(i)) {
688                    return False;
689                }
690                if (!ker.contains(i)) {
691                    entailed = false;
692                }
693            } else if (Boolean.FALSE.equals(constant)) {
694                if (ker.contains(i)) {
695                    return False;
696                }
697                if (env.contains(i)) {
698                    entailed = false;
699                }
700            } else {
701                values = null;
702                entailed = false;
703            }
704        }
705        if (entailed) {
706            return True;
707        }
708        if (values != null) {
709            return equal(set, constant(enumDomain(values)));
710        }
711        return new IrBoolChannel(bools, set, BoolDomain);
712    }

714    public static IrBoolExpr intChannel(IrIntExpr[] ints, IrSetExpr[] sets) {
715        boolean entailed = true;
716        for (int i = 0; i < ints.length; i++) {
717            Integer constant = IrUtil.getConstant(ints[i]);
718            if (constant != null) {
719                IrSetExpr set = sets[constant.intValue()];
720                if (!set.getEnv().contains(i)) {
721                    return False;
722                } else if (!set.getKer().contains(i)) {
723                    entailed = false;
724                }
725            } else {
726                entailed = false;
727            }
728        }
729        if (entailed) {
730            return True;
731        }
732        return new IrIntChannel(ints, sets, BoolDomain);
733    }

787    public static IrBoolExpr sortChannel(IrIntExpr[][] strings, IrIntExpr[] ints) {
788        if (strings.length != ints.length) {
789            throw new IllegalArgumentException();
790        }
791        for (int i = 1; i < strings.length; i++) {
792            if (strings[0].length != strings[i].length) {
793                throw new IllegalArgumentException();
794            }
795        }
796        List<IrBoolExpr> ands = new ArrayList<IrBoolExpr>(0);
797        List<IrIntExpr[]> filterStrings = new ArrayList<IrIntExpr[]>(strings.length);
798        List<IrIntExpr> filterInts = new ArrayList<IrIntExpr>(ints.length);
799        for (int i = 0; i < strings.length; i++) {
800            boolean equivalence = false;
801            for (int j = i + 1; j < strings.length; j++) {
802                if (Arrays.equals(strings[i], strings[j])) {
803                    ands.add(equal(ints[i], ints[j]));
804                    equivalence = true;
805                    break;
806                }
807            }
808            if (!equivalence) {
809                filterStrings.add(strings[i]);
810                filterInts.add(ints[i]);
811            }
812        }
813        if (filterInts.size() == 1) {
814            ands.add(equal(filterInts.get(0), 0));
815        } else if (filterInts.size() > 1) {
816            ands.add(
817                    new IrSortStringsChannel(
818                    filterStrings.toArray(new IrIntExpr[filterStrings.size()][]),
819                    filterInts.toArray(new IrIntExpr[filterInts.size()]),
820                    BoolDomain));
821        }
822        return and(ands);
823    }

832    public static IrBoolExpr selectN(IrBoolExpr[] bools, IrIntExpr n) {
833        if (bools.length == 1) {
834            if (bools[0].equals(n)) {
835                return True;
836            }
837        }
838        boolean entailed = true;
839        IrDomain nDomain = n.getDomain();
840        for (int i = 0; i < bools.length; i++) {
841            Boolean constant = IrUtil.getConstant(bools[i]);
842            if (Boolean.TRUE.equals(constant)) {
843                if (i >= nDomain.getHighBound()) {
844                    return False;
845                }
846                if (i >= nDomain.getLowBound()) {
847                    entailed = false;
848                }
849            } else if (Boolean.FALSE.equals(constant)) {
850                if (i < nDomain.getLowBound()) {
851                    return False;
852                }
853                if (i < nDomain.getHighBound()) {
854                    entailed = false;
855                }
856            } else {
857                entailed = false;
858            }
859        }
860        if (entailed) {
861            return True;
862        }
863        Integer constant = IrUtil.getConstant(n);
864        if (constant != null) {
865            IrBoolExpr[] ands = new IrBoolExpr[bools.length];
866            System.arraycopy(bools, 0, ands, 0, constant.intValue());
867            for (int i = constant.intValue(); i < bools.length; i++) {
868                ands[i] = not(bools[i]);
869            }
870            return and(ands);
871        }
872        return new IrSelectN(bools, n, BoolDomain);
873    }

985        for (IrIntExpr operand : addends) {

1004        int high = domain.getHighBound();

1006            domain = addends[i].getDomain();
1007            low += domain.getLowBound();
1008            high += domain.getHighBound();
1009        }

1173    public static IrIntExpr sum(IrSetExpr set) {
1174        int sum = Util.sum(set.getKer().iterator());
1175        int count = set.getKer().size();
1176
1177        // Calculate low
1178        int low = sum;
1179        int lowCount = count;
1180        TIntIterator envIter = set.getEnv().iterator();
1181        while (lowCount < set.getCard().getHighBound() && envIter.hasNext()) {
1182            int env = envIter.next();
1183            if (env >= 0 && lowCount >= set.getCard().getLowBound()) {
1184                break;
1185            }
1186            if (!set.getKer().contains(env)) {
1187                low += env;
1188                lowCount++;
1189            }
1190        }
1191
1192        // Calculate high
1193        int high = sum;
1194        int highCount = count;
1195        envIter = set.getEnv().iterator(false);
1196        while (highCount < set.getCard().getHighBound() && envIter.hasNext()) {
1197            int env = envIter.next();
1198            if (env <= 0 && highCount >= set.getCard().getLowBound()) {
1199                break;
1200            }
1201            if (!set.getKer().contains(env)) {
1202                high += env;
1203                highCount++;
1204            }
1205        }
1206
1207        return new IrSetSum(set, boundDomain(low, high));

1320    /**
1321     * Relational join.
1322     *
1323     * Union{for all i in take} children[i]
1324     *
1325     * @param take
1326     * @param children
1327     * @return the join expression take.children
1328     */
1329    public static IrSetExpr joinRelation(IrSetExpr take, IrSetExpr[] children, boolean injective) {
1330        if (take.getEnv().isEmpty()) {
1331            return EmptySet;
1332        }
1333        IrSetExpr[] $children = children;
1334        if (take.getEnv().getHighBound() + 1 < $children.length) {
1335            $children = Arrays.copyOf(children, take.getEnv().getHighBound() + 1);
1336        }
1337
1338        int[] constant = IrUtil.getConstant(take);
1339        if (constant != null) {
1340            IrSetExpr[] to = new IrSetExpr[constant.length];
1341            for (int i = 0; i < to.length; i++) {
1342                to[i] = $children[constant[i]];
1343            }
1344            return union(to);
1345        }
1346
1347        // Compute env
1348        TIntIterator iter = take.getEnv().iterator();
1349        IrDomain env;
1350        if (iter.hasNext()) {
1351            IrDomain domain = $children[iter.next()].getEnv();
1352            while (iter.hasNext()) {
1353                domain = IrUtil.union(domain, $children[iter.next()].getEnv());
1354            }
1355            env = domain;
1356        } else {
1357            env = Irs.EmptyDomain;
1358        }
1359
1360        // Compute ker
1361        iter = take.getKer().iterator();
1362        IrDomain ker;
1363        if (iter.hasNext()) {
1364            IrDomain domain = $children[iter.next()].getKer();
1365            while (iter.hasNext()) {
1366                domain = IrUtil.union(domain, $children[iter.next()].getKer());
1367            }
1368            ker = domain;
1369        } else {
1370            ker = Irs.EmptyDomain;
1371        }
1372
1373        // Compute card
1374        IrDomain takeEnv = take.getEnv();
1375        IrDomain takeKer = take.getKer();
1376        IrDomain takeCard = take.getCard();
1377        int index = 0;
1378        int[] childrenLowCards = new int[takeEnv.size() - takeKer.size()];
1379        int[] childrenHighCards = new int[takeEnv.size() - takeKer.size()];
1380        int cardLow = 0, cardHigh = 0;
1381
1382        iter = takeEnv.iterator();
1383        while (iter.hasNext()) {
1384            int val = iter.next();
1385            IrDomain childDomain = $children[val].getCard();
1386            if (takeKer.contains(val)) {
1387                cardLow = injective
1388                        ? cardLow + childDomain.getLowBound()
1389                        : Math.max(cardLow, childDomain.getLowBound());
1390                cardHigh = injective
1391                        ? cardHigh + childDomain.getHighBound()
1392                        : Math.max(cardHigh, childDomain.getHighBound());
1393            } else {
1394                childrenLowCards[index] = childDomain.getLowBound();
1395                childrenHighCards[index] = childDomain.getHighBound();
1396                index++;
1397            }
1398        }
1399        assert index == childrenLowCards.length;
1400        assert index == childrenHighCards.length;
1401
1402        Arrays.sort(childrenLowCards);
1403        Arrays.sort(childrenHighCards);
1404
1405        for (int i = 0; i < takeCard.getLowBound() - takeKer.size(); i++) {
1406            cardLow = injective
1407                    ? cardLow + childrenLowCards[i]
1408                    : Math.max(cardLow, childrenLowCards[i]);
1409        }
1410        for (int i = 0; i < takeCard.getHighBound() - takeKer.size(); i++) {
1411            cardHigh = injective
1412                    ? cardHigh + childrenHighCards[childrenHighCards.length - 1 - i]
1413                    : Math.max(cardHigh, childrenHighCards[childrenHighCards.length - 1 - i]);
1414        }
1415        cardLow = Math.max(cardLow, ker.size());
1416        cardHigh = Math.min(cardHigh, env.size());
1417        IrDomain card = boundDomain(cardLow, cardHigh);
1418
1419        return new IrJoinRelation(take, $children, env, ker, card, injective);
1420    }

1422    public static IrSetExpr joinFunction(IrSetExpr take, IrIntExpr[] refs, Integer globalCardinality) {
1423        int[] constant = IrUtil.getConstant(take);
1424        if (constant != null) {
1425            IrIntExpr[] to = new IrIntExpr[constant.length];
1426            for (int i = 0; i < to.length; i++) {
1427                to[i] = refs[constant[i]];
1428            }
1429            return arrayToSet(to, globalCardinality);
1430        }
1431
1432        // Compute env
1433        TIntIterator iter = take.getEnv().iterator();
1434        IrDomain env;
1435        if (iter.hasNext()) {
1436            IrDomain domain = refs[iter.next()].getDomain();
1437            while (iter.hasNext()) {
1438                domain = IrUtil.union(domain, refs[iter.next()].getDomain());
1439            }
1440            env = domain;
1441        } else {
1442            env = Irs.EmptyDomain;
1443        }
1444
1445        // Compute ker
1446        iter = take.getKer().iterator();
1447        TIntHashSet values = new TIntHashSet(0);
1448        while (iter.hasNext()) {
1449            Integer constantRef = IrUtil.getConstant(refs[iter.next()]);
1450            if (constantRef != null) {
1451                values.add(constantRef.intValue());
1452            }
1453        }
1454        IrDomain ker = values.isEmpty() ? Irs.EmptyDomain : new IrEnumDomain(values.toArray());
1455
1456        // Compute card
1457        IrDomain takeCard = take.getCard();
1458        int lowTakeCard = takeCard.getLowBound();
1459        int highTakeCard = takeCard.getHighBound();
1460        IrDomain card;
1461        if (globalCardinality == null) {
1462            card = lowTakeCard == 0
1463                    ? boundDomain(Math.max(0, ker.size()), Math.min(highTakeCard, env.size()))
1464                    : boundDomain(Math.max(1, ker.size()), Math.min(highTakeCard, env.size()));
1465        } else {
1466            card = boundDomain(
1467                    divRoundUp(Math.max(lowTakeCard, ker.size()), globalCardinality),
1468                    Math.min(highTakeCard, env.size()));
1469        }
1470
1471        return new IrJoinFunction(take, refs, env, ker, card, globalCardinality);
1472    }

1474    private static int divRoundUp(int a, int b) {
1475        assert a >= 0;
1476        assert b > 0;
1477
1478        return (a + b - 1) / b;
1479    }

1481    public static IrSetExpr difference(IrSetExpr minuend, IrSetExpr subtrahend) {
1482        IrDomain env = IrUtil.difference(minuend.getEnv(), subtrahend.getKer());
1483        IrDomain ker = IrUtil.difference(minuend.getKer(), subtrahend.getEnv());
1484        int low = Math.max(0, minuend.getCard().getLowBound() - subtrahend.getCard().getHighBound());
1485        int high = minuend.getCard().getHighBound();
1486        IrDomain card = boundDomain(Math.max(low, ker.size()), Math.min(high, env.size()));
1487        return new IrSetDifference(minuend, subtrahend, env, ker, card);
1488    }

1490    public static IrSetExpr intersection(IrSetExpr... operands) {
1491        switch (operands.length) {
1492            case 0:
1493                return EmptySet;
1494            case 1:
1495                return operands[0];
1496            default:
1497                IrDomain env = IrUtil.intersectionEnvs(operands);
1498                IrDomain ker = IrUtil.intersectionKers(operands);
1499                int low = 0;
1500                int high = operands[0].getCard().getHighBound();
1501                for (int i = 1; i < operands.length; i++) {
1502                    high = Math.max(high, operands[0].getCard().getHighBound());
1503                }
1504                IrDomain card = boundDomain(
1505                        Math.max(low, ker.size()),
1506                        Math.min(high, env.size()));
1507                return new IrSetIntersection(operands, env, ker, card);
1508        }
1509    }

1511    public static IrSetExpr union(IrSetExpr... operands) {
1512        List<IrSetExpr> flatten = new ArrayList<IrSetExpr>(operands.length);
1513        for (IrSetExpr operand : operands) {
1514            flattenUnion(operand, flatten);
1515        }
1516        List<IrSetExpr> filter = new ArrayList<IrSetExpr>();
1517        for (IrSetExpr operand : flatten) {
1518            if (!operand.getEnv().isEmpty()) {
1519                filter.add(operand);
1520            }
1521        }
1522        operands = filter.toArray(new IrSetExpr[filter.size()]);
1523        switch (operands.length) {
1524            case 0:
1525                return EmptySet;
1526            case 1:
1527                return operands[0];
1528            default:
1529                IrDomain env = IrUtil.unionEnvs(operands);
1530                IrDomain ker = IrUtil.unionKers(operands);
1531                IrDomain operandCard = operands[0].getCard();
1532                int low = operandCard.getLowBound();
1533                int high = operandCard.getHighBound();
1534                for (int i = 1; i < operands.length; i++) {
1535                    operandCard = operands[i].getCard();
1536                    low = Math.max(low, operandCard.getLowBound());
1537                    high += operandCard.getHighBound();
1538                }
1539                IrDomain card = boundDomain(
1540                        Math.max(low, ker.size()),
1541                        Math.min(high, env.size()));
1542                return IrUtil.asConstant(new IrSetUnion(operands, env, ker, card));
1543        }
1544    }

1546    private static void flattenUnion(IrSetExpr expr, List<IrSetExpr> flatten) {
1547        if (expr instanceof IrSetUnion) {
1548            IrSetUnion union = (IrSetUnion) expr;
1549            for (IrSetExpr subexpr : union.getOperands()) {
1550                flattenUnion(subexpr, flatten);
1551            }
1552        } else {
1553            flatten.add(expr);
1554        }
1555    }

1557    public static IrSetExpr offset(IrSetExpr set, int offset) {
1558        if (offset == 0) {
1559            return set;
1560        }
1561        int[] constant = IrUtil.getConstant(set);
1562        if (constant != null) {
1563            for (int i = 0; i < constant.length; i++) {
1564                constant[i] += offset;
1565            }
1566            return constant(constant);
1567        }
1568        IrDomain env = IrUtil.offset(set.getEnv(), offset);
1569        IrDomain ker = IrUtil.offset(set.getKer(), offset);
1570        IrDomain card = set.getCard();
1571        return new IrOffset(set, offset, env, ker, card);
1572    }

1574    public static IrSetExpr ternary(IrBoolExpr antecedent, IrSetExpr consequent, IrSetExpr alternative) {
1575        if (IrUtil.isTrue(antecedent)) {
1576            return consequent;
1577        }
1578        if (IrUtil.isFalse(antecedent)) {
1579            return alternative;
1580        }
1581        if (consequent.equals(alternative)) {
1582            return consequent;
1583        }
1584        int[] consequentConstant = IrUtil.getConstant(consequent);
1585        int[] alternativeConstant = IrUtil.getConstant(alternative);
1586        if (consequentConstant != null && alternativeConstant != null && Arrays.equals(consequentConstant, alternativeConstant)) {
1587            return constant(consequentConstant);
1588        }
1589        IrDomain env = IrUtil.union(consequent.getEnv(), alternative.getEnv());
1590        IrDomain ker = IrUtil.intersection(consequent.getKer(), alternative.getKer());
1591        IrDomain card = IrUtil.union(consequent.getCard(), alternative.getCard());
1592        return new IrSetTernary(antecedent, consequent, alternative, env, ker, card);
1593    }
