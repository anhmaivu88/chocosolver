001package org.clafer.ir.analysis;
002
003import org.clafer.ir.IrRewriter;
004import java.util.ArrayList;
005import java.util.HashMap;
006import java.util.HashSet;
007import java.util.Iterator;
008import java.util.List;
009import java.util.Map;
010import java.util.Map.Entry;
011import java.util.Set;
012import org.clafer.ir.IrAnd;
013import org.clafer.ir.IrBoolExpr;
014import org.clafer.ir.IrCard;
015import org.clafer.ir.IrCompare;
016import org.clafer.ir.IrDomain;
017import org.clafer.ir.IrImplies;
018import org.clafer.ir.IrIntExpr;
019import org.clafer.ir.IrIntVar;
020import org.clafer.ir.IrModule;
021import org.clafer.ir.IrSetExpr;
022import org.clafer.ir.IrSetVar;
023import org.clafer.ir.IrUtil;
024import static org.clafer.ir.Irs.*;
025
026/**
027 *
028 * @author jimmy
029 */
030public class Canonicalizer {
031
032    private Canonicalizer() {
033    }
034
035    /**
036     * Rewrites the module in a form that is easier to pattern match and makes
037     * some implicit constraints explicit.
038     *
039     * @param module the module to canonicalize
040     * @return the canonicalized module
041     */
042    public static IrModule canonical(IrModule module) {
043        CanonicalRewriter rewriter = new CanonicalRewriter();
044
045        IrModule optModule = rewriter.rewriteAndNonNops(module, null);
046        for (Entry<IrSetVar, IrIntVar> entry : rewriter.setVarCards.entrySet()) {
047            optModule.addConstraint(equal(entry.getValue(), card(entry.getKey())));
048        }
049        rewriter.setVars.removeAll(rewriter.setVarCards.keySet());
050        for (IrSetVar setVar : rewriter.setVars) {
051            if (setVar.getCard().getLowBound() > setVar.getKer().size()
052                    || setVar.getCard().getHighBound() < setVar.getEnv().size()) {
053                // These variables need to have their cardinalities constrainted.
054                optModule.addConstraint(equal(
055                        domainInt("|" + setVar.getName() + "|", setVar.getCard()),
056                        card(setVar)));
057            }
058        }
059        return optModule;
060    }
061
062    private static class CanonicalRewriter extends IrRewriter<Void> {
063
064        private final Set<IrSetVar> setVars = new HashSet<IrSetVar>();
065        private final Map<IrSetVar, IrIntVar> setVarCards = new HashMap<IrSetVar, IrIntVar>();
066
067        @Override
068        public IrBoolExpr visit(IrAnd ir, Void a) {
069            Map<IrBoolExpr, Set<IrBoolExpr>> implications = new HashMap<IrBoolExpr, Set<IrBoolExpr>>();
070            List<IrBoolExpr> opts = new ArrayList<IrBoolExpr>(ir.getOperands().length);
071            for (IrBoolExpr operand : ir.getOperands()) {
072                IrBoolExpr opt = rewrite(operand, a);
073                if (opt instanceof IrImplies) {
074                    IrImplies implies = (IrImplies) opt;
075                    Set<IrBoolExpr> consequents = implications.get(implies.getAntecedent());
076                    if (consequents == null) {
077                        consequents = new HashSet<IrBoolExpr>();
078                        implications.put(implies.getAntecedent(), consequents);
079                    }
080                    consequents.add(implies.getConsequent());
081                } else {
082                    opts.add(opt);
083                }
084            }
085            for (Map.Entry<IrBoolExpr, Set<IrBoolExpr>> entry : implications.entrySet()) {
086                IrBoolExpr antecedent = entry.getKey();
087                Set<IrBoolExpr> consequents = entry.getValue();
088                if (!consequents.isEmpty()) {
089                    IrBoolExpr notAntecedent = not(antecedent);
090                    Set<IrBoolExpr> alternatives = implications.get(notAntecedent);
091                    if (alternatives != null && !alternatives.isEmpty()) {
092                        Iterator<IrBoolExpr> iter = consequents.iterator();
093                        while (iter.hasNext()) {
094                            IrBoolExpr consequent = iter.next();
095                            IrBoolExpr alternative = not(consequent);
096                            if (alternatives.remove(alternative)) {
097                                opts.add(
098                                        // Either choice is fine, but prefer to pick
099                                        // the antecedent that is not negative because
100                                        // it looks nicer and some cases easier to
101                                        // propagate.
102                                        antecedent.isNegative()
103                                        ? ifOnlyIf(notAntecedent, alternative)
104                                        : ifOnlyIf(antecedent, consequent));
105                                iter.remove();
106                            }
107                        }
108                    }
109                    for (IrBoolExpr consequent : consequents) {
110                        opts.add(implies(antecedent, consequent));
111                    }
112                    consequents.clear();
113                }
114            }
115            return and(opts);
116        }
117
118        @Override
119        public IrBoolExpr visit(IrCompare ir, Void a) {
120            IrIntExpr left = rewrite(ir.getLeft(), a);
121            IrIntExpr right = rewrite(ir.getRight(), a);
122            if (ir.getOp().isEquality()) {
123                IrDomain leftDomain = left.getDomain();
124                IrDomain rightDomain = right.getDomain();
125                // This canoicalizes the form somewhat so that is easier for the other
126                // analysis to find pattern matches.
127                if (leftDomain.size() == 2) {
128                    // Prefer the one that's closer to 0.
129                    int prefer = leftDomain.getLowBound();
130                    int unprefer = leftDomain.getHighBound();
131                    if (Math.abs(leftDomain.getLowBound()) > Math.abs(leftDomain.getHighBound())) {
132                        prefer = leftDomain.getHighBound();
133                        unprefer = leftDomain.getLowBound();
134                    }
135                    Integer rightConstant = IrUtil.getConstant(right);
136                    if (rightConstant != null && rightConstant.intValue() == unprefer) {
137                        return compare(left, ir.getOp().negate(), prefer);
138                    }
139                }
140                if (rightDomain.size() == 2) {
141                    // Prefer the one that's closer to 0.
142                    int prefer = rightDomain.getLowBound();
143                    int unprefer = rightDomain.getHighBound();
144                    if (Math.abs(rightDomain.getLowBound()) > Math.abs(rightDomain.getHighBound())) {
145                        prefer = rightDomain.getHighBound();
146                        unprefer = rightDomain.getLowBound();
147                    }
148                    Integer leftConstant = IrUtil.getConstant(left);
149                    if (leftConstant != null && leftConstant.intValue() == unprefer) {
150                        return compare(prefer, ir.getOp().negate(), right);
151                    }
152                }
153            }
154            return compare(left, ir.getOp(), right);
155        }
156
157        @Override
158        public IrIntExpr visit(IrCard ir, Void a) {
159            /*
160             * |Literal| is a very common expression that is used very often.
161             * Instead, rewrite as NewVar = |Literal| so that every reference
162             * to |Literal| can use the same variable.
163             */
164            IrSetExpr set = rewrite(ir.getSet(), a);
165            if (set instanceof IrSetVar) {
166                IrSetVar setVar = (IrSetVar) set;
167                IrIntVar card = setVarCards.get(setVar);
168                if (card == null) {
169                    card = domainInt("|" + setVar.getName() + "|", setVar.getCard());
170                    setVarCards.put(setVar, card);
171                }
172                return card;
173            }
174            return card(set);
175        }
176
177        @Override
178        public IrSetExpr visit(IrSetVar ir, Void a) {
179            setVars.add(ir);
180            return ir;
181        }
182    };
183}
