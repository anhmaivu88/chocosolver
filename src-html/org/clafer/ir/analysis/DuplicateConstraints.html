001package org.clafer.ir.analysis;
002
003import java.util.HashSet;
004import java.util.Set;
005import org.clafer.ir.IrBoolExpr;
006import org.clafer.ir.IrModule;
007
008/**
009 * Due to coalescing, it is possible to end up with multiple cardinality
010 * variables for the same set. Remove the duplicates.
011 *
012 * @author jimmy
013 */
014public class DuplicateConstraints {
015
016    private DuplicateConstraints() {
017    }
018
019    public static IrModule removeDuplicates(IrModule module) {
020        Set<IrBoolExpr> constraints = new HashSet<IrBoolExpr>(module.getConstraints());
021        return new IrModule().addConstraints(constraints);
022    }
023}
