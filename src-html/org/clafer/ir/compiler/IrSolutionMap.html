001package org.clafer.ir.compiler;
002
003import java.util.Map;
004import org.clafer.ir.IrBoolConstant;
005import org.clafer.ir.IrBoolVar;
006import org.clafer.ir.IrIntConstant;
007import org.clafer.ir.IrIntVar;
008import org.clafer.ir.IrSetConstant;
009import org.clafer.ir.IrSetVar;
010import solver.Solver;
011import solver.variables.BoolVar;
012import solver.variables.IntVar;
013import solver.variables.SetVar;
014
015/**
016 * Maps IR non-constant variables to their translated Choco variables. The
017 * get&lt;Type&gt;Var methods fail on constants. The get&lt;Type&gt;Value
018 * methods handles constants, and is the preferred way of lookup.
019 *
020 * @author jimmy
021 */
022public class IrSolutionMap {
023
024    private final Solver solver;
025    public final Map<IrIntVar, IrIntVar> coalescedIntVars;
026    public final Map<IrIntVar, IntVar> intVars;
027    private final Map<IrSetVar, IrSetVar> coalescedSetVars;
028    private final Map<IrSetVar, SetVar> setVars;
029
030    IrSolutionMap(
031            Solver solver,
032            Map<IrIntVar, IrIntVar> coalescedIntVars,
033            Map<IrIntVar, IntVar> intVars,
034            Map<IrSetVar, IrSetVar> coalescedSetVars,
035            Map<IrSetVar, SetVar> setVars) {
036        this.solver = solver;
037        this.coalescedIntVars = coalescedIntVars;
038        this.intVars = intVars;
039        this.coalescedSetVars = coalescedSetVars;
040        this.setVars = setVars;
041    }
042
043    public BoolVar getBoolVar(IrBoolVar var) {
044        IrBoolVar boolVar = (IrBoolVar) coalescedIntVars.get(var);
045        if (boolVar == null) {
046            boolVar = var;
047        }
048        if (boolVar instanceof IrBoolConstant) {
049            return ((IrBoolConstant) boolVar).getValue() ? solver.ONE : solver.ZERO;
050        }
051        return (BoolVar) intVars.get(boolVar);
052    }
053
054    public boolean getBoolValue(IrBoolVar var) {
055        IrBoolVar boolVar = (IrBoolVar) coalescedIntVars.get(var);
056        if (boolVar == null) {
057            boolVar = var;
058        }
059        if (boolVar instanceof IrBoolConstant) {
060            return ((IrBoolConstant) boolVar).getValue();
061        }
062        return intVars.get(boolVar).getValue() != 0;
063    }
064
065    public BoolVar[] getBoolVars(IrBoolVar... vars) {
066        BoolVar[] bvars = new BoolVar[vars.length];
067        for (int i = 0; i < bvars.length; i++) {
068            bvars[i] = getBoolVar(vars[i]);
069        }
070        return bvars;
071    }
072
073    public boolean[] getBoolValues(IrBoolVar... vars) {
074        boolean[] bvalues = new boolean[vars.length];
075        for (int i = 0; i < bvalues.length; i++) {
076            bvalues[i] = getBoolValue(vars[i]);
077        }
078        return bvalues;
079    }
080
081    public IntVar getIntVar(IrIntVar var) {
082        IrIntVar intVar = coalescedIntVars.get(var);
083        if (intVar == null) {
084            intVar = var;
085        }
086        return intVars.get(intVar);
087    }
088
089    public int getIntValue(IrIntVar var) {
090        IrIntVar intVar = coalescedIntVars.get(var);
091        if (intVar == null) {
092            intVar = var;
093        }
094        if (intVar instanceof IrIntConstant) {
095            return ((IrIntConstant) intVar).getValue();
096        }
097        return intVars.get(intVar).getValue();
098    }
099
100    public IntVar[] getIntVars(IrIntVar... vars) {
101        IntVar[] ivars = new IntVar[vars.length];
102        for (int i = 0; i < ivars.length; i++) {
103            ivars[i] = getIntVar(vars[i]);
104        }
105        return ivars;
106    }
107
108    public int[] getIntValues(IrIntVar... vars) {
109        int[] ivalues = new int[vars.length];
110        for (int i = 0; i < ivalues.length; i++) {
111            ivalues[i] = getIntValue(vars[i]);
112        }
113        return ivalues;
114    }
115
116    public SetVar getSetVar(IrSetVar var) {
117        IrSetVar setVar = coalescedSetVars.get(var);
118        if (setVar == null) {
119            setVar = var;
120        }
121        return setVars.get(setVar);
122    }
123
124    public int[] getSetValue(IrSetVar var) {
125        IrSetVar setVar = coalescedSetVars.get(var);
126        if (setVar == null) {
127            setVar = var;
128        }
129        if (setVar instanceof IrSetConstant) {
130            return ((IrSetConstant) setVar).getValue();
131        }
132        return setVars.get(setVar).getValue();
133    }
134
135    public SetVar[] getSetVars(IrSetVar... vars) {
136        SetVar[] svars = new SetVar[vars.length];
137        for (int i = 0; i < svars.length; i++) {
138            svars[i] = getSetVar(vars[i]);
139        }
140        return svars;
141    }
142
143    public int[][] getSetValues(IrSetVar... vars) {
144        int[][] svalues = new int[vars.length][];
145        for (int i = 0; i < svalues.length; i++) {
146            svalues[i] = getSetValue(vars[i]);
147        }
148        return svalues;
149    }
150}
